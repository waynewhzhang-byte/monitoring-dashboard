```bash

# 3. 导入数据库

# 4. 运行 Prisma 迁移
npx prisma generate
npx prisma migrate deploy

# 启动主大屏 Web 服务
pm2 start npm --name "monitoring-web" -- start

# 启动后台数据采集服务 (非常重要：负责与 OpManager 同步数据)
# 生产环境使用 node 运行编译后的 js 文件或通过 ts-node 运行
pm2 start "npx ts-node -r tsconfig-paths/register --project tsconfig.node.json src/services/collector/start.ts" --name "monitoring-collector"
```
